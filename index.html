🚀 Supabase Setup Guide

Run this SQL in your Supabase project (SQL Editor) to create the required tables:

-- 1. Profiles (one per user)
create table profiles (
  id uuid references auth.users on delete cascade primary key,
  first_name text, last_name text,
  title text, company text,
  email text, phone text, bio text,
  links jsonb default '[]',
  username text unique,
  updated_at timestamptz default now()
);

-- 2. Leads (many per user)
create table leads (
  id uuid default gen_random_uuid() primary key,
  user_id uuid references auth.users on delete cascade,
  name text not null, company text, email text, phone text,
  notes text, status text default 'new',
  enriched_data jsonb,
  created_at timestamptz default now()
);

-- 3. Row Level Security
alter table profiles enable row level security;
alter table leads enable row level security;

create policy "Own profile" on profiles for all using (auth.uid() = id);
create policy "Own leads" on leads for all using (auth.uid() = user_id);
